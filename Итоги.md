// 1. Получаем настройки
let rawSort = dv.current().sort_mode;
let sortMode = String(rawSort).replace(/['"]/g, "").trim();

const pages = dv.pages('"Тесты"').where(p => p.mechanics_world != null);

// 2. Функция-художник
function createBar(percent, width = "70px", span = "span") {
    let color = "#ef4444"; // Красный
    let bg = "rgba(239, 68, 68, 0.2)";
    
    if (percent >= 80) {
        color = "#22c55e"; // Зеленый
        bg = "rgba(34, 197, 94, 0.2)";
    } else if (percent >= 50) {
        color = "#eab308"; // Желтый
        bg = "rgba(234, 179, 8, 0.2)";
    }
    
    return `<div style="display:flex; align-items:center; gap:6px;"><div style="width:${width}; height:12px; background-color:${bg}; border-radius:4px; overflow:hidden;"><div style="width:${percent}%; height:100%; background-color:${color}; transition:width 0.3s;"></div></div><${span}>${percent}%</${span}></div>`;
}

// 3. Расчет данных
let data = pages.map(p => {
    const safeSum = (arr) => arr.reduce((a, b) => (a || 0) + (b || 0), 0);
    
    const mech = Math.round(safeSum([p.mechanics_world, p.mechanics_terrain_score, p.mechanics_blocks_display, p.mechanics_blocks_break, p.mechanics_blocks_install, p.mechanics_drop, p.mechanics_autopickup, p.mechanics_inventory, p.mechanics_walk, p.mechanics_look]) / 12 * 100);
    const vis = Math.round(safeSum([(p.visual_crosshair ? 1 : 0), p.visual_block_under_crosshair, p.visual_deep, p.visual_ui_quality, p.visual_blocks_quality, p.visual_blocks_variability, p.visual_feedback]) / 11 * 100);
    const opt = Math.round(safeSum([p.opt_errors, p.opt_performance, p.opt_performance_chunk_loading]) / 7 * 100);
    const total = Math.round(((mech * 12 + vis * 11 + opt * 7) / 30));
    
    return {
        link: p.file.link,
        name: p.file.name,
        company: String(p.company || "-"),
        mech: mech,
        vis: vis,
        opt: opt,
        total: total
    };
});

data = [...data];

// 4. Сортировка
if (sortMode === "rating_best") data.sort((a, b) => b.total - a.total);
else if (sortMode === "rating_worst") data.sort((a, b) => a.total - b.total);
else if (sortMode === "opt_best") data.sort((a, b) => b.opt - a.opt);
else if (sortMode === "vis_best") data.sort((a, b) => b.vis - a.vis);
else if (sortMode === "mech_best") data.sort((a, b) => b.mech - a.mech);
else if (sortMode === "name_az") data.sort((a, b) => a.name.localeCompare(b.name));
else if (sortMode === "company_az") data.sort((a, b) => a.company.localeCompare(b.company));
else data.sort((a, b) => b.total - a.total);

// 5. Отрисовка
const headers = ["🤖 Модель", "🏢 Компания", "⚙️ Мех", "🎨 Виз", "🚀 Опт", "🏆 Итог"];

const rows = data.map(d => {
    return [
        d.link,      
        d.company,
        createBar(d.mech, "50px"),  
        createBar(d.vis, "50px"),   
        createBar(d.opt, "50px"),   
        createBar(d.total, "200px", "b") 
    ];
});

dv.table(headers, rows);
